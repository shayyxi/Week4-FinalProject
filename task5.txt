library(dplyr)
library(ggplot2)
NEI <- readRDS("summarySCC_PM25.rds")
SCC <- readRDS("Source_Classification_Code.rds")
list<-grep("Mobile.*Vehicles",SCC$EI.Sector)
list2<-SCC[list,]
list3<-list2$SCC
task4<-subset(NEI,SCC %in% list3)
task5<-subset(task4,fips=="24510")
df<-tbl_df(task5)
task4<-group_by(df,year)
a<-summarize(task4,total=sum(Emissions))
s<-as.data.frame(a)
png("plot5.png")
ggplot(s,aes(factor(year),total))+geom_bar(stat="identity",fill="black")
dev.off()
